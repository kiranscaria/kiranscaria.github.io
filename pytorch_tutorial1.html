Converting numpy array into torch tensor

In [72]:
import numpy as np
      a = np.ones(5)
      b = torch.from_numpy(a)
      np.add(a, 1, out=a)
      print(a)
      print(b)

Cuda Tensors¶

Tensors can be moved onto GPU using .cuda function

In [74]:
if torch.cuda.is_available():
          x = x.cuda()
          y = y.cuda()
          print(x + y)
      else:
          print("CUDA not found.\nExecuting nonetheless........\n")
          print(x + y)
